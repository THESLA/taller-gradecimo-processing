Ejemplo 1: El círculo curioso

mouseX mouseY interacción
Un círculo que te sigue 
void setup() {
  size(400, 400);
}

void draw() {
  background(20, 20, 40);
  
  // mouseX y mouseY siguen la posición del mouse
  fill(231, 76, 60);
  ellipse(mouseX, mouseY, 60, 60);
  
  // Círculo más pequeño que persigue al grande
  fill(243, 156, 18, 150);
  ellipse(mouseX - 30, mouseY - 30, 30, 30);
}
mouseX = posición X del mouse mouseY = posición Y del mouse Se actualiza automáticamente en draw()

Ejemplo 2: Pinta con clics

mousePressed random if
Haz clic para pintar círculos de colores 
// Variables para guardar datos
int x = 0, y = 0;
boolean pintar = false;

void setup() {
  size(400, 400);
  background(20, 20, 40);
}

void draw() {
  if (pintar) {
    // random() da un valor aleatorio
    fill(random(255), random(255), random(255), 150);
    ellipse(x, y, random(20, 60), random(20, 60));
    pintar = false;
  }
}

void mousePressed() {
  // Se ejecuta UNA VEZ cada vez que haces clic
  x = mouseX;
  y = mouseY;
  pintar = true;
}
mousePressed() se ejecuta al hacer clic random(255) = número aleatorio entre 0 y 255 Las variables guardan información

Ejemplo 3: Pelota que rebota

variables animación if velocidad
La pelota se mueve sola y rebota en los bordes 
// Variables que cambian en el tiempo
float x = 200;
float y = 200;
float velX = 3;
float velY = 2;

void setup() {
  size(400, 400);
}

void draw() {
  background(20, 20, 40);
  
  // Dibuja la pelota
  fill(46, 204, 113);
  ellipse(x, y, 40, 40);
  
  // Mueve la pelota
  x = x + velX;
  y = y + velY;
  
  // Rebota en los bordes con if
  if (x > width - 20 || x < 20) {
    velX = -velX;  // Invierte dirección
  }
  if (y > height - 20 || y < 20) {
    velY = -velY;
  }
}
float = número con decimales x = x + velX mueve la figura if + rebote = simulación física

Ejemplo 4: Arte generativo interactivo

todo combinado keyPressed random mousePressed
Presiona cualquier tecla para reiniciar 
// Arte generativo: figuras que bailan
float t = 0;

void setup() {
  size(400, 400);
  colorMode(HSB, 360, 100, 100);
}

void draw() {
  background(0, 0, 10);
  
  // Figuras que se mueven con seno (ondas)
  for (int i = 0; i < 10; i++) {
    float angulo = t + i * 0.5;
    float x = 200 + sin(angulo) * 150;
    float y = 200 + cos(angulo) * 150;
    
    fill(i * 36, 80, 90, 70);
    ellipse(x, y, 30 + mouseX / 20, 30 + mouseY / 20);
  }
  
  t = t + 0.02;
  
  // Texto interactivo
  fill(255);
  text("Mouse: tamaño | Clic: nuevo color | Tecla: reinicia", 40, 380);
}

void mousePressed() {
  // Reinicia con colores diferentes
  colorMode(random(3) < 1 ? RGB : HSB, 
             random(255), random(255), random(255));
}

void keyPressed() {
  // keyPressed() detecta cualquier tecla
  setup();  // Reinicia todo
}
for + sin/cos = movimiento circular keyPressed() responde al teclado mouseX/mouseY controlan tamaño

Crea tu obra de arte interactiva

Usando todo lo que aprendimos (día 1 + día 2), crea tu propia composición interactiva original. El tema es libre, pero debe incluir:

2+ figuras diferentes

Colores

mouseX / mouseY

Un if

Que cambie algo según una condición
